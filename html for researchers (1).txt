                <pre><code>def parse_fasta(filename):
    with open(filename) as f:
        seqs = {}
        seq_name = ''
        for line in f:
            line = line.strip()
            if line.startswith('&gt;'):
                seq_name = line[1:]
                seqs[seq_name] = ''
            else:
                seqs[seq_name] += line
        return seqs
</code></pre>
